export const CodeTabs = ({ tabs, defaultTab }) => {
  if (!tabs || tabs.length === 0) return null;

  const [activeTab, setActiveTab] = useState(defaultTab || tabs[0].id);
  const currentTab = tabs.find((tab) => tab.id === activeTab) || tabs[0];

  return (
    <div className="w-full rounded-2xl bg-[#0A0A0A] border border-white/10 shadow-2xl overflow-hidden flex flex-col">
      
      {/* Header Row */}
      <div className="flex items-center px-4 pt-3 bg-[#111113] border-b border-white/5 shrink-0">
        
        {/* Sleek Mac Dots (Slightly smaller for minimalism) */}
        <div className="flex gap-2 mr-6 sm:mr-8 mb-3 opacity-80 shrink-0">
          <div className="w-2.5 h-2.5 rounded-full bg-[#FF5F56] border border-[#E0443E]/50"></div>
          <div className="w-2.5 h-2.5 rounded-full bg-[#FFBD2E] border border-[#DEA123]/50"></div>
          <div className="w-2.5 h-2.5 rounded-full bg-[#27C93F] border border-[#1AAB29]/50"></div>
        </div>

        {/* Navigation Tabs */}
        <div className="flex gap-5 sm:gap-6 text-[13px] font-mono text-zinc-500 mt-1 overflow-x-auto scrollbar-hide">
          {tabs.map((tab) => {
            const isActive = activeTab === tab.id;
            return (
              <button
                key={tab.id}
                onClick={() => setActiveTab(tab.id)}
                className={`pb-3 whitespace-nowrap transition-colors border-b-2 outline-none ${
                  isActive
                    ? "text-zinc-100 border-indigo-500 font-sans"
                    : "border-transparent hover:text-zinc-300"
                }`}
              >
                {tab.label}
              </button>
            );
          })}
        </div>

        {/* Language Indicator (Right aligned) */}
        <div className="ml-auto mb-2 hidden sm:block text-[10px] font-mono text-zinc-600 uppercase tracking-widest shrink-0">
          {currentTab.language || "code"}
        </div>
      </div>

      {/* Code Body */}
      <div className="p-5 sm:p-6 overflow-y-auto flex-1 max-h-[500px]">
        <pre className="font-mono text-[13px] sm:text-sm leading-[1.8] text-zinc-200 m-0 p-0 bg-transparent">
          <code>{currentTab.code}</code>
        </pre>
      </div>
      
    </div>
  );
};
